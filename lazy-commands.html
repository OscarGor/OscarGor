    match_id: "FB3200",
    date: "2026-02-05T02:00",
    home_team: "馬德里CFF女足",
    away_team: "特內里費女足",
    competition: "女子西班牙盃",
    qimen_data: {
        pattern: "陽遁3局",
        palaces: { /* 九宮數據 */ },
        gods: { home: "值符", away: "值使" }
    },
    actual_result: {
        final_score: "2-1",
        corners: { home: 5, away: 3 }
    }
})
#2. 批量導出數據
導出所有歷史數據用於備份
AI.exportAllData({
    format: "json",
    include: ["matches", "ai_params", "patterns"],
    filename: `qimen_football_backup_${new Date().toISOString().slice(0,10)}`
})
